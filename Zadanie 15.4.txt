SELECT I.*, IL.NAME AS ISSUES_LIST_NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.issueslist_id = IL.ID;

SELECT I.*, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.ASSIGNED_TO_USER_ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.ID) AS TASKS_ASSIGNED
FROM USERS U
JOIN ISSUES I ON U.ID = I.ASSIGNED_TO_USER_ID
GROUP BY U.ID, U.FIRSTNAME, U.LASTNAME;
